# Using a try/except block as a function to use multiple times with ease.

def int_check(prompt):
    while True:
        try:
            return int(input(prompt))
        except ValueError:
            print("That is not a valid entry")

# Just "plug and play" with different prompts that are inputed as integers

int_check("What is the year of your vehicle:")
int_check("What is your age?")
int_check("what is the temperature?")

# Using a try/except block as a function with a min_value and a max_value range.

def int_check(prompt, min_value, max_value):
    while True:
        try:
            value = int(input(prompt))
            # logic check for the range
            if min_value <= value <= max_value:
                return value
            else:
                print(f"Invalid input, Please enter a 4 digit year {min_value} - {max_value}")
        except ValueError:
            print("Invalid input, Please enter a number")

# Also "plug and play". 

car_year = int_check("What year is your vehicle?")
print(f"Car year saved {car_year}")
